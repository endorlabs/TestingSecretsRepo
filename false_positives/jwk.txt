# --- 3. Schema / type definitions (field names, not key material) ---

// TypeScript interface -- field names and types, not key material
interface JsonWebKey {
  kty: string;
  crv?: string;
  alg?: string;
  kid?: string;
  n?: string;
  e?: string;
  x?: string;
  y?: string;
  d?: string; // private key component (a type, never a real value here)
}
